CREATE STREAM st_dept_enriched
WITH(KAFKA_TOPIC='st_dept_enriched', VALUE_FORMAT='JSON')

========================================For JDBC Sink connector

--need to create another stream for st_dept_enriched, because kafka topics have plain json, but jdbc sink connector, requires a schema in the msg
ksql>
CREATE STREAM st_dept_data_2_db_jdbc_snk (
    DEPARTMENT_ID INT,
    DEPARTMENT_NAME STRING,
    CHANGED_DATE STRING,
    NEW_COLUMN1 STRING,
    NEW_COLUMN2 STRING,
    COMPANY_NAME STRING,
    PROCESSED_AT STRING
) WITH (
    KAFKA_TOPIC='st_dept_enriched',
    VALUE_FORMAT='JSON'
);

--this creates a topic in kafka, with schema in the messages
ksql>
CREATE STREAM st_dept_data_2_db_for_sink_topic
WITH (
    KAFKA_TOPIC = 'st_dept_data_2_db_for_sink_topic',
    VALUE_FORMAT='JSON_SR',
    PARTITIONS=1
) AS 
SELECT 
    DEPARTMENT_ID AS "dept_id", 
    DEPARTMENT_NAME AS "dept_name", 
    CHANGED_DATE AS "changed_at", 
    NEW_COLUMN1 AS "col1", 
    NEW_COLUMN2 AS "col2", 
    COMPANY_NAME AS "company", 
    FORMAT_TIMESTAMP(FROM_UNIXTIME(UNIX_TIMESTAMP(PROCESSED_AT, 'yyyy-MM-dd HH:mm:ss')), 'yyyy-MM-dd HH:mm:ss') AS "processed_at"
FROM 
    st_dept_data_2_db_jdbc_snk;

--to see the changes live in the csv_data enriched stream
SELECT * FROM st_dept_data_2_db_for_sink_topic EMIT CHANGES;

==============================================Updating policy streams

--to drop the schema registry file for stream
curl -X DELETE http://localhost:8081/subjects/kt_dbz_policies_proc_json_sr-value

--to collect raw data from debezium topic

CREATE STREAM st_policy_raw_data (
    BEFORE STRUCT<"tran_id" INT, "status" STRING, "policy_name" string, "created_at" string>,
    AFTER STRUCT<"tran_id" INT, "status" STRING, "policy_name" string, "created_at" string>,
    "op" string
) WITH (
    KAFKA_TOPIC = 'kt_dbz_policy_raw_data.public.policies_mst',
    VALUE_FORMAT = 'JSON',
    PARTITIONS = 1
);

--to create stream for jdbc sink to insert / update the data]
CREATE STREAM st_policy_proc_data 
with(
    VALUE_FORMAT='JSON_SR',
    PARTITIONS = 1,
    KAFKA_TOPIC='kt_dbz_policies_proc_json_sr'
)
AS
    SELECT  
        AFTER->"tran_id" as "tran_id",
        AFTER->"status" as "status",
        AFTER->"policy_name" as "policy_name",
        AFTER->"created_at" as "created_at",
        TIMESTAMPTOSTRING(ROWTIME, 'yyyy-MM-dd HH:mm:ss') as "processed_at",
        "op" as "op_type"
    FROM
        st_policy_raw_data
    WHERE
        AFTER IS NOT NULL;

====================================Streams as per Events

--Policy created
CREATE STREAM st_evt_policy_created
WITH (
    KAFKA_TOPIC='topic_event_policy_created',
    VALUE_FORMAT='JSON_SR',
    PARTITIONS=1
) AS
SELECT 
    AFTER->"tran_id" AS "policy_tran_id",
    AFTER->"policy_name" AS "policy_name",
    AFTER->"status" AS "status",
    AFTER->"created_at" AS "created_at",
    TIMESTAMPTOSTRING(ROWTIME, 'yyyy-MM-dd HH:mm:ss') AS "event_timestamp"
FROM 
    st_policy_raw_data
WHERE 
    "op" = 'c';


-----Policy_lapsed
CREATE STREAM st_evt_policy_lapsed
WITH (
    KAFKA_TOPIC='topic_event_policy_lapsed',
    VALUE_FORMAT='JSON_SR',
    PARTITIONS=1
) AS
SELECT 
    AFTER->"tran_id" AS "policy_tran_id",
    BEFORE->"status" AS "old_status",
    AFTER->"status" AS "new_status",
    TIMESTAMPTOSTRING(ROWTIME, 'yyyy-MM-dd HH:mm:ss') AS "lapse_timestamp"
FROM 
    st_policy_raw_data
WHERE 
    "op" = 'u'
    AND LCASE(AFTER->"status") = 'lapse' 
    AND LCASE(BEFORE->"status") = 'inforce';


---------Policy Inforced
CREATE STREAM st_event_policy_inforce
WITH (
    KAFKA_TOPIC='topic_event_policy_inforce',
    VALUE_FORMAT='JSON_SR',
    PARTITIONS=1
) AS
SELECT 
    AFTER->"tran_id" AS "policy_tran_id",
    BEFORE->"status" AS "old_status",
    AFTER->"status" AS "new_status",
    TIMESTAMPTOSTRING(ROWTIME, 'yyyy-MM-dd HH:mm:ss') AS "processed_at"
FROM 
    st_policy_raw_data
WHERE 
    LCASE(AFTER->"status") = 'inforce' 
    AND (BEFORE->"status" IS NULL OR LCASE(BEFORE->"status") != 'inforce');

WITH (
