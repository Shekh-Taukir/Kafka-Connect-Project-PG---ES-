--create a stream (just like table in sql) for existing topic
ksql> CREATE STREAM st_dept_stream (
department_id INT,
department_name VARCHAR,
changed_date VARCHAR,
new_column1 VARCHAR,
new_column2 VARCHAR) WITH (
KAFKA_TOPIC = 'department-transfer-data-v1',
VALUE_FORMAT = 'JSON');

--create a enchanced stream to make updated data
ksql>
CREATE STREAM st_dept_enriched
WITH(KAFKA_TOPIC='st_dept_enriched', VALUE_FORMAT='JSON')
AS SELECT *,
'MyGlobal-Corp-IN' AS company_name,
TIMESTAMPTOSTRING(ROWTIME, 'yyyy-MM-dd HH:mm:ss') as processed_at
FROM st_dept_stream;

--to see the changes live in the enriched stream
SELECT * FROM st_dept_enriched EMIT CHANGES;

--create a stream for existing topic for csv_file topic
ksql>
CREATE STREAM st_csv_raw_data_stream(
    tran_id  VARCHAR,
    first_name VARCHAR,
    last_name VARCHAR,
    email VARCHAR,
    age VARCHAR,
    city VARCHAR,
    created_at VARCHAR
) WITH (
    KAFKA_TOPIC = 'csv_raw_data',
    VALUE_FORMAT='JSON'
);

--create a enchanced stream to make updated data for csv raw data stream
ksql>
CREATE STREAM 
    st_csv_2_data_enriched
WITH(KAFKA_TOPIC='kt_csv_2_data_enrch', VALUE_FORMAT='JSON') AS 
SELECT 
    *,
    CONCAT('emp_', first_name) as emp_id,
    'Global_Work-IN' AS company_name,
    TIMESTAMPTOSTRING(ROWTIME, 'yyyy-MM-dd HH:mm:ss') as processed_at
FROM 
    st_csv_raw_data_stream;

--to see the changes live in the csv_data enriched stream
SELECT * FROM st_csv_2_data_enriched EMIT CHANGES;


========================================For JDBC Sink connector

--need to create another stream for st_dept_enriched, because kafka topics have plain json, but jdbc sink connector, requires a schema in the msg
ksql>
CREATE STREAM st_dept_data_2_db_jdbc_snk (
    DEPARTMENT_ID INT,
    DEPARTMENT_NAME STRING,
    CHANGED_DATE STRING,
    NEW_COLUMN1 STRING,
    NEW_COLUMN2 STRING,
    COMPANY_NAME STRING,
    PROCESSED_AT STRING
) WITH (
    KAFKA_TOPIC='st_dept_enriched',
    VALUE_FORMAT='JSON'
);

--this creates a topic in kafka, with schema in the messages
ksql>
CREATE STREAM st_dept_data_2_db_for_sink_topic
WITH (
    KAFKA_TOPIC = 'st_dept_data_2_db_for_sink_topic',
    VALUE_FORMAT='JSON_SR',
    PARTITIONS=1
) AS 
SELECT 
    DEPARTMENT_ID AS "dept_id", 
    DEPARTMENT_NAME AS "dept_name", 
    CHANGED_DATE AS "changed_at", 
    NEW_COLUMN1 AS "col1", 
    NEW_COLUMN2 AS "col2", 
    COMPANY_NAME AS "company", 
    FORMAT_TIMESTAMP(FROM_UNIXTIME(UNIX_TIMESTAMP(PROCESSED_AT, 'yyyy-MM-dd HH:mm:ss')), 'yyyy-MM-dd HH:mm:ss') AS "processed_at"
FROM 
    st_dept_data_2_db_jdbc_snk;

--to see the changes live in the csv_data enriched stream
SELECT * FROM st_dept_data_2_db_for_sink_topic EMIT CHANGES;
